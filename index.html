// Base64 轉換成 ArrayBuffer function base64ToArrayBuffer(base64) { const binaryString = window.atob(base64); const len = binaryString.length; const bytes = new Uint8Array(len); for (let i = 0; i < len; i++) { bytes[i] = binaryString.charCodeAt(i); } return bytes.buffer; } // 將 PCM 16-bit 音訊封裝進 WAV 容器，以便瀏覽器直接播放 function pcmToWav(pcm16, sampleRate) { const buffer = new ArrayBuffer(44 + pcm16.length * 2); const view = new DataView(buffer); // RIFF 識別碼 writeString(view, 0, 'RIFF'); // 檔案長度 view.setUint32(4, 36 + pcm16.length * 2, true); // RIFF 格式類型 writeString(view, 8, 'WAVE'); // 格式區塊識別碼 writeString(view, 12, 'fmt '); // 格式區塊長度 view.setUint32(16, 16, true); // 音訊編碼格式 (1 表示無壓縮的 PCM) view.setUint16(20, 1, true); // 聲道數 (單聲道) view.setUint16(22, 1, true); // 採樣率 view.setUint32(24, sampleRate, true); // 每秒位元組率 (採樣率 * 聲道數 * 每個採樣位元組數) view.setUint32(28, sampleRate * 2, true); // 區塊對齊 (聲道數 * 每個採樣位元組數) view.setUint16(32, 2, true); // 每個採樣位元數 view.setUint16(34, 16, true); // 數據區塊識別碼 writeString(view, 36, 'data'); // 數據區塊長度 view.setUint32(40, pcm16.length * 2, true); // 寫入原始 PCM 音訊數據 for (let i = 0; i < pcm16.length; i++) { view.setInt16(44 + i * 2, pcm16[i], true); } return new Blob([buffer], { type: 'audio/wav' }); } function writeString(view, offset, string) { for (let i = 0; i < string.length; i++) { view.setUint8(offset + i, string.charCodeAt(i)); } } function calculateScientificTargets(type, weight, status) { const w = Number(weight); if (!w || isNaN(w) || w <= 0) return { rer: 0, calories: 0, water: 0 };
